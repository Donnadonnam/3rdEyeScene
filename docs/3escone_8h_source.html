00060   inline float Cone::angle() const
00061   {
00062     return scale().x;
00063     // scale X/Y encode the radius of the cone base.
00064     // Convert to angle angle as:
00065     //   tan(theta) = radius / length

00067     const Vector3f s = scale();

00069     const float radius = s.x;
