39 // By Bob Jenkins, 1996. bob_jenkins@burtleburtle.net.
40 #define VHASH_JENKINS_MIX(a, b, c) \
41  a -= b; a -= c; a ^= (c>>13); \
42  b -= c; b -= a; b ^= (a<<8); \
43  c -= a; c -= b; c ^= (b>>13); \
44  a -= b; a -= c; a ^= (c>>12); \
45  b -= c; b -= a; b ^= (a<<16); \
46  c -= a; c -= b; c ^= (b>>5); \
47  a -= b; a -= c; a ^= (c>>3); \
48  b -= c; b -= a; b ^= (a<<10); \
49  c -= a; c -= b; c ^= (b>>15);
