00039 // By Bob Jenkins, 1996. bob_jenkins@burtleburtle.net.
00040 #define VHASH_JENKINS_MIX(a, b, c) \
00041   a -= b; a -= c; a ^= (c>>13); \
00042   b -= c; b -= a; b ^= (a<<8);  \
00043   c -= a; c -= b; c ^= (b>>13); \
00044   a -= b; a -= c; a ^= (c>>12); \
00045   b -= c; b -= a; b ^= (a<<16); \
00046   c -= a; c -= b; c ^= (b>>5);  \
00047   a -= b; a -= c; a ^= (c>>3);  \
00048   b -= c; b -= a; b ^= (a<<10); \
00049   c -= a; c -= b; c ^= (b>>15);
