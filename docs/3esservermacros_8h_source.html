72 #define TES_CATEGORY(server, _name, _categoryId, _parentId, _active) \
73  { \
74  tes::CategoryNameMessage msg; \
75  msg.categoryId = _categoryId; \
76  msg.parentId = _parentId; \
77  msg.defaultActive = (_active) ? 1 : 0; \
78  const size_t nameLen = (_name) ? strlen(_name) : 0u; \
79  msg.nameLength = (uint16_t)((nameLen <= 0xffffu) ? nameLen : 0xffffu); \
80  msg.name = _name; \
81  tes::sendMessage((server), tes::MtCategory, tes::CategoryNameMessage::MessageId, msg); \
82  }

140 #define TES_SERVER_UPDATE(server, ...) \
141  { \
142  (server).updateFrame(__VA_ARGS__); \
143  tes::ConnectionMonitor *_conMon = (server).connectionMonitor(); \
144  if (_conMon->mode() == tes::ConnectionMonitor::Synchronous) \
145  { \

147  } \
148  _conMon->commitConnections(); \

150  }
