77 #define TES_CATEGORY(server, _name, _categoryId, _parentId, _active) \
78  { \
79  tes::CategoryNameMessage msg; \
80  msg.categoryId = _categoryId; \
81  msg.parentId = _parentId; \
82  msg.defaultActive = (_active) ? 1 : 0; \
83  const size_t nameLen = (_name) ? strlen(_name) : 0u; \
84  msg.nameLength = (uint16_t)((nameLen <= 0xffffu) ? nameLen : 0xffffu); \
85  msg.name = _name; \
86  tes::sendMessage((server), tes::MtCategory, tes::CategoryNameMessage::MessageId, msg); \
87  }
